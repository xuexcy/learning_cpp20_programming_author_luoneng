// coroutine_handle , future, promise, 协程帧 frame(coroutine_state)的关系
┌─────────────────────────────────────────────────────┐
│                    用户代码层                          │
│  ┌─────────────────────────────────────────────┐    │
│  │  auto future = coroutine();  // 调用协程      │    │
│  │  future.resume();            // 恢复协程      │    │
│  └─────────────────────────────────────────────┘    │
└─────────────────────────────────────────────────────┘
                           │
                           ▼
┌─────────────────────────────────────────────────────┐
│                  Future (协程返回对象)                │
│  ┌─────────────────────────────────────────────┐    │
│  │ • 用户可见的接口                               │    │
│  │ • 持有 coroutine_handle                       │    │
│  │ • 控制协程生命周期                              │    │
│  │ • 示例：Task<T>, Generator<T>                 │    │
│  └─────────────────────────────────────────────┘    │
└─────────────────────────────────────────────────────┘
                           │
                           ▼
┌─────────────────────────────────────────────────────┐
│           coroutine_handle (协程句柄)                │
│  ┌─────────────────────────────────────────────┐    │
│  │ • 轻量级句柄（通常是一个指针）                   │    │
│  │ • 指向协程帧 (frame*)                          │    │
│  │ • 提供 resume()、destroy() 等操作              │    │
│  │ • 可以访问关联的 Promise                        │    │
│  └─────────────────────────────────────────────┘    │
└─────────────────────────────────────────────────────┘
                           │
                           ▼
┌─────────────────────────────────────────────────────┐
│           协程帧 frame (coroutine_state)            │
│  ┌─────────────────────────────────────────────┐    │
│  │  ┌─────────────────────────────────────┐   │    │
│  │  │  函数指针表                           │   │    │
│  │  │  • void (*resume)(frame*)           │   │    │
│  │  │  • void (*destroy)(frame*)          │   │    │
│  │  └─────────────────────────────────────┘   │    │
│  │  ┌─────────────────────────────────────┐   │    │
│  │  │  Promise 对象                         │   │    │
│  │  │  • 管理协程内部逻辑                      │   │    │
│  │  │  • 处理 co_return/co_yield             │   │    │
│  │  │  • 存储异常信息                         │   │    │
│  │  └─────────────────────────────────────┘   │    │
│  │  ┌─────────────────────────────────────┐   │    │
│  │  │  协程局部变量                          │   │    │
│  │  │  • 参数副本                            │   │    │
│  │  │  • 局部变量                            │   │    │
│  │  │  • 挂起点信息                           │   │    │
│  │  └─────────────────────────────────────┘   │    │
│  └─────────────────────────────────────────────┘    │
└─────────────────────────────────────────────────────┘
